<script>

var FINNA = [

  {id:"ekirjasto", name:"e-Kirjasto", icon:"&#128215;", color:"#1a5f3f", type:"Free &middot; Finnish national digital library",

   filter:"filter%5B%5D=building%3A%220%2FNLF%2F%22&filter%5B%5D=online_boolean%3A%221%22",

   link:function(q){return "https://e-kirjasto.fi/en/search?query="+encodeURIComponent(q)}},

  {id:"helmet", name:"Helmet", icon:"&#128218;", color:"#b5451b", type:"Free &middot; Helsinki metro area libraries",

   filter:"filter%5B%5D=building%3A%220%2FHelmet%2F%22",

  {id:"helle", name:"Helle", icon:"&#128217;", color:"#2255a4", type:"Free &middot; P&auml;ij&auml;t-H&auml;me region libraries",

   filter:"filter%5B%5D=building%3A%220%2FHelle%2F%22",

   link:function(q){return "https://helle.finna.fi/Search/Results?lookfor="+encodeURIComponent(q)+"&type=AllFields"}}

];


var LINKS = [

  {name:"Audible",  icon:"&#127911;", color:"#f29d12", type:"Purchase/subscription", link:function(q){return "https://www.audible.com/search?keywords="+encodeURIComponent(q)}},

  {name:"BookBeat", icon:"&#127897;", color:"#e63a4a", type:"Subscription",          link:function(q){return "https://www.bookbeat.com/search?query="+encodeURIComponent(q)}},

  {name:"Storytel", icon:"&#128214;", color:"#6b4fbb", type:"Subscription",          link:function(q){return "https://www.storytel.com/search?q="+encodeURIComponent(q)}}

];


var busy = false;


function esc(s) {

  return String(s||"").replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;");

}


function searchFinna(src, query, cb) {

  var url = "https://api.finna.fi/api/v1/search?lookfor="+encodeURIComponent(query)+"&"+src.filter+"&field%5B%5D=title&field%5B%5D=authors&field%5B%5D=year&field%5B%5D=formats&limit=3&lng=en";

  fetch(url)

    .then(function(r){ if(!r.ok) throw new Error("HTTP "+r.status); return r.json(); })

    .then(function(d){

      if(d.status !== "OK") throw new Error("API error");

      var recs = (d.records||[]).map(function(r){

        var author = "";

        if(r.authors && r.authors.primary) author = Object.keys(r.authors.primary)[0];

        else if(r.nonPresenterAuthors && r.nonPresenterAuthors[0]) author = r.nonPresenterAuthors[0].name;

        return {title:r.title||"", author:author, year:r.year||"", format:(r.formats&&r.formats[0]&&r.formats[0].translated)||""};

      });

      cb(null, recs);

    })

    .catch(function(e){ cb(e, null); });

}


function renderCard(src, state, query) {

  var loading = state.loading, error = state.error, results = state.results;

  var found = results && results.length > 0;

  var border = (loading||!found) ? "#e0d8d0" : src.color;


  var badge = "";

  if(loading) badge = '<span class="searching">Searching...</span>';

  else if(error) badge = '<span class="err">Error</span>';

  else if(found) badge = '<a href="'+src.link(query)+'" target="_blank" class="found" style="background:'+src.color+'">Found &rarr;</a>';

  else badge = '<span class="notfound">Not found</span>';


  var rhtml = "";

  if(found) {

    rhtml = '<div class="rlist">';

    for(var i=0;i<results.length;i++) {

      var r = results[i];

      var meta = [r.author, r.year, r.format].filter(Boolean).join(" &middot; ");

      rhtml += '<a href="'+src.link(query)+'" target="_blank" class="ritem"><div class="rt">'+esc(r.title)+'</div>'+(meta?'<div class="rm">'+meta+'</div>':'')+'</a>';

    }

    rhtml += '<a href="'+src.link(query)+'" target="_blank" class="vall" style="color:'+src.color+'">View all &rarr;</a></div>';

  }


  return '<div class="card" id="c-'+src.id+'" style="border-left-color:'+border+'"><div class="ch"><span class="ci">'+src.icon+'</span><div style="flex:1"><div class="cn">'+src.name+'</div><div class="ct">'+src.type+'</div></div>'+badge+'</div>'+rhtml+'</div>';

}


function updateCard(src, state, query) {

  var old = document.getElementById("c-"+src.id);

  if(!old) return;

  var tmp = document.createElement("div");

  tmp.innerHTML = renderCard(src, state, query);

  old.parentNode.replaceChild(tmp.firstChild, old);

}


function go() {

  var q = document.getElementById("q").value.trim();

  if(!q || busy) return;

  busy = true;

  document.getElementById("btn").textContent = "Searching...";

  document.getElementById("btn").disabled = true;


  var html = '<p style="color:#5a3e28;margin-bottom:16px;font-size:.9rem">Results for <strong>'+esc(q)+'</strong>:</p>';

  html += '<div class="lbl">&#127467;&#127470; Finnish Libraries &mdash; live via Finna API</div>';

  for(var i=0;i<FINNA.length;i++) html += renderCard(FINNA[i], {loading:true}, q);

  html += '<div class="lbl">&#127911; Streaming &amp; Purchase</div>';

  for(var j=0;j<LINKS.length;j++) {

    var s = LINKS[j];

    html += '<a href="'+s.link(q)+'" target="_blank" class="lcard" style="border-left:5px solid '+s.color+'"><span class="ci">'+s.icon+'</span><div style="flex:1"><div class="cn">'+s.name+'</div><div class="ct">'+s.type+'</div></div><span class="larr" style="color:'+s.color+'">Search &rarr;</span></a>';

  }

  document.getElementById("main").innerHTML = html;


  var done = 0;

  for(var k=0;k<FINNA.length;k++) {

    (function(src){

      searchFinna(src, q, function(err, results){

        if(err) updateCard(src, {loading:false, error:true, results:null}, q);

        else updateCard(src, {loading:false, error:null, results:results}, q);

        done++;

        if(done === FINNA.length) {

          busy = false;

          document.getElementById("btn").textContent = "Search";

          document.getElementById("btn").disabled = false;

        }

      });

    })(FINNA[k]);

  }

}


document.getElementById("q").addEventListener("keydown", function(e){ if(e.key==="Enter") go(); });
